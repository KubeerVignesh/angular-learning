String Format

{{" Name : " + title }}

{{" LowerCase : " + title | lowercase }}

{{" upperCase : " + title | uppercase }}

{{" titleCase : " + title | titlecase }}

{{" startFrom : " + title | slice:2 }}

{{" Json Format : " + (personalDetails | json) }}

Number Format

{{ 5.678 | number : '1.2-3'}}

{{ 5.678 | number : '2.3-4'}}

{{ 5.678 | number : '2.4-5'}}

{{ 5.678 | number : '2.1-2'}}

Percentage Format

{{0.25 | percent}}

Currency Format

{{25 | currency}}

{{25 | currency : 'GBP'}}

{{.25 | currency : 'EUR'}}

{{100000000 | currency : 'INR'}}

{{date | date : 'shortDate'}}

{{date | date : 'shortTime'}}

{{date | date : 'mediumDate'}}

{{date | date : 'mediumTime'}}

{{date | date: 'long'}}

{{date | date: 'longDate'}}
